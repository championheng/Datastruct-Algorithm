 1 void Bublesort(int a[],int n)
 2 {
 3      int i,j,k;
 4      for(j=0;j<n-1;j++)   /* 气泡法要排序n次*/
 5      {
 6           for(i=0;i<n-j-1;i++)  /* 值比较大的元素沉下去后，只把剩下的元素中的最大值再沉下去就可以啦 */
 7           {
 8                if(a[i]>a[i+1])  /* 把值比较大的元素沉到底 */
 9                {
10                     k=a[i];
11                     a[i]=a[i+1];
12                     a[i+1]=k;
13                }
14           }
15      }
16 }
